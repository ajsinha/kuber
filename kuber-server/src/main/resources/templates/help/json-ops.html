JSON Operations

Store, retrieve, update, and search JSON documents with powerful query operators including regex matching. Kuber Extension commands.

Command Reference
CommandSyntaxDescription
JSETJSET key json [path] [ttl]Store JSON document at key
JGETJGET key [path]Get JSON document or value at path
JUPDATEJUPDATE key json [ttl]Merge/update JSON (upsert with deep merge)
JREMOVEJREMOVE key ["attr1", ...]Remove attributes from JSON document
JDELJDEL key [path]Delete JSON document or path within it
JSEARCHJSEARCH querySearch JSON by field values with operators
Aliases: JSON.SET, JSON.GET, JSON.UPDATE, JSON.REMOVE, JSON.DEL, JSON.SEARCH are also supported.
JSET / JSON.SET - Store JSON Document

Store a JSON document at a key. Optionally specify a path within existing JSON and TTL.

  • Redis CLI
  • Python
  • Java
  • C#
# Store complete JSON document
JSET user:1001 {"name":"Alice","age":30,"city":"NYC"}

# Store at specific path (root)
JSET user:1001 {"name":"Alice","age":30} $

# Store with TTL (3600 seconds)
JSET session:abc {"token":"xyz","user":"alice"} $ 3600

# Store nested JSON
JSET order:5001 {"id":"5001","items":[{"sku":"A1","qty":2},{"sku":"B2","qty":1}],"total":150.00}
with KuberRedisClient('localhost', 6380, api_key='kub_xxx') as client:
    # Store JSON document
    user = {"name": "Alice", "age": 30, "city": "NYC"}
    client.json_set('user:1001', user)
    
    # Store with TTL
    session = {"token": "xyz", "user": "alice"}
    client.json_set('session:abc', session, ttl=3600)
    
    # Store nested JSON
    order = {
        "id": "5001",
        "items": [{"sku": "A1", "qty": 2}, {"sku": "B2", "qty": 1}],
        "total": 150.00
    }
    client.json_set('order:5001', order)
try (KuberClient client = new KuberClient("localhost", 6380, "kub_xxx")) {
    // Store JSON document
    String userJson = """
        {"name":"Alice","age":30,"city":"NYC"}
        """;
    client.jsonSet("user:1001", userJson);
    
    // Store with TTL (3600 seconds)
    client.jsonSet("session:abc", "{\"token\":\"xyz\",\"user\":\"alice\"}", "$", 3600);
    
    // Store nested JSON
    String orderJson = """
        {"id":"5001","items":[{"sku":"A1","qty":2}],"total":150.00}
        """;
    client.jsonSet("order:5001", orderJson);
}
using var client = new KuberClient("localhost", 6380, "kub_xxx");

// Store JSON document
var user = new { name = "Alice", age = 30, city = "NYC" };
await client.JsonSetAsync("user:1001", user);

// Store with TTL
var session = new { token = "xyz", user = "alice" };
await client.JsonSetAsync("session:abc", session, ttl: 3600);

// Store nested JSON
var order = new {
    id = "5001",
    items = new[] { new { sku = "A1", qty = 2 } },
    total = 150.00
};
await client.JsonSetAsync("order:5001", order);
JGET / JSON.GET - Get JSON Document

Retrieve a JSON document or a specific path within it using JSONPath syntax.

  • Redis CLI
  • Python
  • Java
  • C#
# Get entire document
JGET user:1001
# Returns: {"name":"Alice","age":30,"city":"NYC"}

# Get specific field
JGET user:1001 $.name
# Returns: "Alice"

# Get nested field
JGET order:5001 $.items[0].sku
# Returns: "A1"

# Get array element
JGET order:5001 $.items[0]
# Returns: {"sku":"A1","qty":2}
with KuberRedisClient('localhost', 6380, api_key='kub_xxx') as client:
    # Get entire document
    user = client.json_get('user:1001')
    print(f"User: {user}")  # {'name': 'Alice', 'age': 30, ...}
    
    # Get specific field
    name = client.json_get('user:1001', '$.name')
    print(f"Name: {name}")  # "Alice"
    
    # Get nested field
    sku = client.json_get('order:5001', '$.items[0].sku')
    print(f"SKU: {sku}")  # "A1"
try (KuberClient client = new KuberClient("localhost", 6380, "kub_xxx")) {
    // Get entire document
    JsonNode user = client.jsonGet("user:1001");
    System.out.println("User: " + user);
    
    // Get specific field
    JsonNode name = client.jsonGet("user:1001", "$.name");
    System.out.println("Name: " + name.asText());  // "Alice"
    
    // Get nested field
    JsonNode sku = client.jsonGet("order:5001", "$.items[0].sku");
    System.out.println("SKU: " + sku.asText());
}
using var client = new KuberClient("localhost", 6380, "kub_xxx");

// Get entire document as typed object
var user = await client.JsonGetAsync("user:1001");
Console.WriteLine($"User: {user.Name}");

// Get specific field
var name = await client.JsonGetAsync("user:1001", "$.name");
Console.WriteLine($"Name: {name}");  // "Alice"

// Get nested field
var sku = await client.JsonGetAsync("order:5001", "$.items[0].sku");
Console.WriteLine($"SKU: {sku}");
JUPDATE / JSON.UPDATE - Merge JSON v1.7.2

Update/merge JSON document. Creates if key doesn't exist (upsert). Performs deep merge on existing JSON objects.

  • Redis CLI
  • Python
  • Java
  • C#
# Original: {"name":"Alice","age":30}
JSET user:1001 {"name":"Alice","age":30}

# Merge new fields (add city, update age)
JUPDATE user:1001 {"age":31,"city":"NYC","email":"alice@example.com"}

# Result: {"name":"Alice","age":31,"city":"NYC","email":"alice@example.com"}
JGET user:1001

# Upsert - creates if not exists
JUPDATE user:9999 {"name":"NewUser","status":"active"}
with KuberRedisClient('localhost', 6380, api_key='kub_xxx') as client:
    # Set initial JSON
    client.json_set('user:1001', {"name": "Alice", "age": 30})
    
    # Merge/update JSON
    updated = client.json_update('user:1001', {
        "age": 31,
        "city": "NYC",
        "email": "alice@example.com"
    })
    print(f"Updated: {updated}")
    # {'name': 'Alice', 'age': 31, 'city': 'NYC', 'email': '...'}
    
    # Upsert - creates if not exists
    client.json_update('user:9999', {"name": "NewUser"})
try (KuberClient client = new KuberClient("localhost", 6380, "kub_xxx")) {
    // Set initial JSON
    client.jsonSet("user:1001", "{\"name\":\"Alice\",\"age\":30}");
    
    // Merge/update JSON
    JsonNode updated = client.jsonUpdate("user:1001", 
        "{\"age\":31,\"city\":\"NYC\",\"email\":\"alice@example.com\"}");
    System.out.println("Updated: " + updated);
    
    // Upsert - creates if not exists
    client.jsonUpdate("user:9999", "{\"name\":\"NewUser\"}");
}
using var client = new KuberClient("localhost", 6380, "kub_xxx");

// Set initial JSON
await client.JsonSetAsync("user:1001", new { name = "Alice", age = 30 });

// Merge/update JSON
var updated = await client.JsonUpdateAsync("user:1001", new {
    age = 31,
    city = "NYC",
    email = "alice@example.com"
});
Console.WriteLine($"Updated: {updated.Name}, {updated.Age}");

// Upsert - creates if not exists
await client.JsonUpdateAsync("user:9999", new { name = "NewUser" });
JREMOVE / JSON.REMOVE - Remove Attributes v1.7.2

Remove specified attributes from a JSON document. Does nothing if key doesn't exist or value is not a JSON object.

  • Redis CLI
  • Python
  • Java
  • C#
# Original: {"name":"Alice","age":30,"temp":"x","debug":true}
JSET user:1001 {"name":"Alice","age":30,"temp":"x","debug":true}

# Remove single attribute
JREMOVE user:1001 ["temp"]
# Result: {"name":"Alice","age":30,"debug":true}

# Remove multiple attributes
JREMOVE user:1001 ["age","debug"]
# Result: {"name":"Alice"}

JGET user:1001
with KuberRedisClient('localhost', 6380, api_key='kub_xxx') as client:
    # Set JSON with extra fields
    client.json_set('user:1001', {
        "name": "Alice", "age": 30, "temp": "x", "debug": True
    })
    
    # Remove single attribute
    result = client.json_remove('user:1001', ['temp'])
    print(f"After remove temp: {result}")
    
    # Remove multiple attributes
    result = client.json_remove('user:1001', ['age', 'debug'])
    print(f"After remove age,debug: {result}")
    # {'name': 'Alice'}
try (KuberClient client = new KuberClient("localhost", 6380, "kub_xxx")) {
    // Set JSON with extra fields
    client.jsonSet("user:1001", 
        "{\"name\":\"Alice\",\"age\":30,\"temp\":\"x\",\"debug\":true}");
    
    // Remove using varargs
    JsonNode result = client.jsonRemove("user:1001", "temp");
    System.out.println("After remove temp: " + result);
    
    // Remove multiple attributes
    result = client.jsonRemove("user:1001", "age", "debug");
    System.out.println("After remove: " + result);
    
    // Or use List
    result = client.jsonRemove("user:1001", Arrays.asList("field1", "field2"));
}
using var client = new KuberClient("localhost", 6380, "kub_xxx");

// Set JSON with extra fields
await client.JsonSetAsync("user:1001", new {
    name = "Alice", age = 30, temp = "x", debug = true
});

// Remove single attribute
var result = await client.JsonRemoveAsync("user:1001", "temp");
Console.WriteLine($"After remove temp: {result}");

// Remove multiple attributes
result = await client.JsonRemoveAsync("user:1001", "age", "debug");
Console.WriteLine($"After remove: {result}");
// {"name":"Alice"}
JDEL / JSON.DEL - Delete JSON

Delete a JSON document or a specific path within it.

  • Redis CLI
  • Python
  • Java
  • C#
# Delete specific path within JSON
JSET user:1001 {"name":"Alice","age":30,"temp":"x"}
JDEL user:1001 $.temp
JGET user:1001
# Returns: {"name":"Alice","age":30}

# Delete entire document
JDEL user:1001
JGET user:1001
# Returns: (nil)
with KuberRedisClient('localhost', 6380, api_key='kub_xxx') as client:
    client.json_set('user:1001', {"name": "Alice", "age": 30, "temp": "x"})
    
    # Delete specific path
    client.json_delete('user:1001', '$.temp')
    print(client.json_get('user:1001'))  # {'name': 'Alice', 'age': 30}
    
    # Delete entire document
    client.json_delete('user:1001')
    print(client.json_get('user:1001'))  # None
try (KuberClient client = new KuberClient("localhost", 6380, "kub_xxx")) {
    client.jsonSet("user:1001", "{\"name\":\"Alice\",\"age\":30,\"temp\":\"x\"}");
    
    // Delete specific path
    client.jsonDel("user:1001", "$.temp");
    System.out.println(client.jsonGet("user:1001"));
    
    // Delete entire document
    client.jsonDel("user:1001");
}
using var client = new KuberClient("localhost", 6380, "kub_xxx");
await client.JsonSetAsync("user:1001", new { name = "Alice", age = 30, temp = "x" });

// Delete specific path
await client.JsonDeleteAsync("user:1001", "$.temp");

// Delete entire document
await client.JsonDeleteAsync("user:1001");
JSEARCH / JSON.SEARCH - Search JSON Documents

Search JSON documents using field-based queries with comparison, regex, and IN clause operators.

v1.8.2: Added IN clause support for matching multiple values per attribute.
Supported Operators
OperatorDescriptionExample
=Equalsdepartment=Engineering
=[v1|v2]IN clause (match any)status=[active|pending]
!=Not equalsstatus!=inactive
>Greater thansalary>50000
>=Greater or equalage>=21
<Less thanprice<100
<=Less or equalquantity<=10
~=Regex matchname~=^J.*
IN Clause Syntax (v1.8.2)

Use [value1|value2|value3] to match any of multiple values:

PatternDescription
field=[v1|v2|v3]Match if field equals v1 OR v2 OR v3
field1=[a|b],field2=[x|y]Match if field1 IN (a,b) AND field2 IN (x,y)
status=[active|pending],country=[USA|UK|CA]Multiple attributes with IN clauses
  • Redis CLI
  • Python
  • Java
  • C#
# Setup test data
JSET emp:1 {"name":"Alice","dept":"Engineering","salary":95000,"country":"USA"}
JSET emp:2 {"name":"Bob","dept":"Sales","salary":75000,"country":"UK"}
JSET emp:3 {"name":"John","dept":"Engineering","salary":85000,"country":"CA"}
JSET emp:4 {"name":"Jane","dept":"HR","salary":70000,"country":"USA"}

# Search by single field
JSEARCH dept=Engineering
# Returns: [emp:1, emp:3] with values

# IN clause - match multiple values (v1.8.2)
JSEARCH dept=[Engineering|Sales]
# Returns: [emp:1, emp:2, emp:3]

# IN clause on multiple attributes (v1.8.2)
JSEARCH dept=[Engineering|Sales],country=[USA|UK]
# Returns: [emp:1, emp:2] (Engineering/USA and Sales/UK)

# Search with comparison
JSEARCH salary>80000
# Returns: [emp:1, emp:3]

# Multiple conditions (AND)
JSEARCH dept=Engineering,salary>90000
# Returns: [emp:1]

# Regex search - names starting with J
JSEARCH name~=^J.*
# Returns: [emp:3, emp:4]

# Combine IN clause with other operators
JSEARCH dept=[Engineering|HR],salary>=70000
# Returns: [emp:1, emp:3, emp:4]
with KuberRedisClient('localhost', 6380, api_key='kub_xxx') as client:
    # Setup test data
    client.json_set('emp:1', {"name": "Alice", "dept": "Engineering", "salary": 95000, "country": "USA"})
    client.json_set('emp:2', {"name": "Bob", "dept": "Sales", "salary": 75000, "country": "UK"})
    client.json_set('emp:3', {"name": "John", "dept": "Engineering", "salary": 85000, "country": "CA"})
    
    # Search by department
    results = client.json_search('dept=Engineering')
    for r in results:
        print(f"{r['key']}: {r['value']}")
    
    # IN clause - multiple values for one field (v1.8.2)
    results = client.json_search('dept=[Engineering|Sales]')
    print(f"Engineering or Sales: {len(results)}")
    
    # IN clause on multiple attributes (v1.8.2)
    results = client.json_search('dept=[Engineering|Sales],country=[USA|UK]')
    
    # Using convenience method with dict (v1.8.2)
    conditions = {
        "dept": ["Engineering", "Sales"],
        "country": ["USA", "UK", "CA"]
    }
    results = client.json_search_in(conditions)
    
    # Multiple conditions
    results = client.json_search('dept=Engineering,salary>90000')
    
    # Regex search
    results = client.json_search('name~=^J.*')
    print(f"Names starting with J: {len(results)}")
try (KuberClient client = new KuberClient("localhost", 6380, "kub_xxx")) {
    // Setup test data
    client.jsonSet("emp:1", "{\"name\":\"Alice\",\"dept\":\"Engineering\",\"salary\":95000}");
    client.jsonSet("emp:2", "{\"name\":\"Bob\",\"dept\":\"Sales\",\"salary\":75000}");
    client.jsonSet("emp:3", "{\"name\":\"John\",\"dept\":\"Engineering\",\"salary\":85000}");
    
    // Search by department
    List<JsonNode> results = client.jsonSearch("dept=Engineering");
    for (JsonNode r : results) {
        System.out.println(r);
    }
    
    // IN clause - multiple values for one field (v1.8.2)
    results = client.jsonSearch("dept=[Engineering|Sales]");
    
    // IN clause on multiple attributes (v1.8.2)
    results = client.jsonSearch("dept=[Engineering|Sales],country=[USA|UK]");
    
    // Using convenience method with Map (v1.8.2)
    Map<String, List<String>> conditions = new LinkedHashMap<>();
    conditions.put("dept", Arrays.asList("Engineering", "Sales"));
    conditions.put("country", Arrays.asList("USA", "UK", "CA"));
    results = client.jsonSearchIn(conditions);
    
    // Multiple conditions
    results = client.jsonSearch("dept=Engineering,salary>90000");
    
    // Regex search
    results = client.jsonSearch("name~=^J.*");
}
